34static std::string readToken(std::istream& is) {
35 std::string token;

93static bool readHeader(std::istream& is, PNM_Header& header, bool hasMax = false) {
94 const int32_t MAX_PIXEL_VALUE = 255;
95
96 try
97 {
98 std::string format = readToken(is); // throws
99 int32_t width = std::stoi(readToken(is)); // throws
100 int32_t height = std::stoi(readToken(is)); // throws
101
102 uint32_t maxValue;
103 if (hasMax) maxValue = std::stoi(readToken(is)); // throws
104 if (hasMax && maxValue > MAX_PIXEL_VALUE) return false;
105
106 header = { format, width, height, hasMax ? maxValue : 1 };
107 }
108 catch (const std::exception&) { return false; }
109
110 return true;
111}

160static void createFilenameWithTimestamp(std::string& name, const std::string filename) {
161
162 if (name == NO_SAVE_NAME) {
163 name = filename;
164 std::string time = getCurrentTime();
165
166 // find dot index
167 std::size_t dotPos = name.rfind('.'); // doesn't throw
168 if (dotPos == std::string::npos) {
169 name = name + "_" + time; // no dot

171
172 // we have a valid dot position
